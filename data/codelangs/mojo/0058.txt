@compiler.register("my_add")
struct MyAdd:
    @staticmethod
    def execute(
        output: OutputTensor,
        x: InputTensor[dtype=output.dtype, rank=output.rank, ...],
        y: InputTensor[dtype=output.dtype, rank=output.rank, ...],
    ):
        output[0] = x[0] + y[0]

    @staticmethod
    def shape(
        x: InputTensor,
        y: InputTensor,
    ) raises -> IndexList[x.rank]:
        raise "NotImplemented"
